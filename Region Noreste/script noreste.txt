ROUTER EL CABRITO
enable
configure terminal
hostname ElCabrito

! Seguridad y SSH 
ip domain-name juanmark.com
crypto key generate rsa modulus 1024
username admin secret cisco
line vty 0 4
 transport input ssh
 login local
exit
enable secret cisco
service password-encryption

! Interfaz hacia la Nube 
! IP del enlace: 148.60.255.8/30, Usamos la .9
interface Serial0/0/0
 ip address 148.60.255.9 255.255.255.252
 clock rate 128000
 no shutdown
exit

! Interfaz hacia el Switch
interface GigabitEthernet0/1
 no shutdown
exit
interface GigabitEthernet0/1.200
 encapsulation dot1Q 200
 ip address 148.60.200.3 255.255.255.248
 
 ! Configuración HSRP 
 standby 1 ip 148.60.200.1
 standby 1 priority 150
 standby 1 preempt
 no shutdown
exit

! Ruteo RIP
router rip
 version 2
 no auto-summary
 network 148.60.0.0
 passive-interface GigabitEthernet0/1.200
exit
